1    import java.util.ArrayList;
2    
3    /**
4     * This class represents a Used Car Lot
5     *
6     * @author Michael Miller
7     */
8    public class UsedCarLot {
9    
10       /** The Cars in the used car lot */
11       private ArrayList<Car> inventory;
12   
13       /**
14        * Instantiates a UsedCarLot object and
15        * initializes the inventory to an empty ArrayList
16        */
17       public UsedCarLot() {
18           inventory = new ArrayList<Car>();
19       }
20   
21       /**
22        * Returns the inventory of the used car lot
23        *
24        * @return The inventory
25        */
26       public ArrayList<Car> getInventory() {
27           return inventory;
28       }
29   
30       /**
31        * Adds a Car to the end of the inventory
32        *
33        * @param carToAdd the Car to add
34        */
35       public void addCar(Car carToAdd) {
36           inventory.add(carToAdd);
37       }
38   
39       /**
40        * Swaps the Car at idx1 with the Car at idx2 in inventory and returns true;
41        * however, if either idx1 or idx2 exceed the bounds of the inventory or is negative,
42        * no changes are made to the inventory and false is returned.
43        *
44        * @param idx1 The index of the first Car to swap
45        * @param idx2 The index of the second Car to swap
46        * @return True if the swap was successful, false if the swap was not successful
47        */
48       public boolean swap(int idx1, int idx2) {
49           if (idx1 < inventory.size() && idx2 < inventory.size() && idx1 >= 0 && idx2 >= 0) {
50               Car car1 = inventory.get(idx1);
51               Car car2 = inventory.get(idx2);
52               inventory.set(idx2, car1);
53               inventory.set(idx1, car2);
54               return true;
55           } else {
56               //do not change inventory and return false
57               return false;
58           }
59       }
60   
61       /**
62        * Adds a Car to the inventory list at the index specified
63        * by indexToAdd; this method increases the size of inventory by 1
64        * <p>
65        * PRECONDITION: 0 &lt;= indexToAdd &lt; inventory.size()
66        *
67        * @param indexToAdd The index in the inventory to add the new Car
68        * @param carToAdd The Car to add
69        */
70       public void addCar(int indexToAdd, Car carToAdd)
71       {
72           // uses the other ArrayList "add" method that takes two parameters
73           inventory.add(indexToAdd, carToAdd);
74       }
75   
76       /**
77        * "Sells" the Car located at indexOfCarToSell which
78        * removes it from the inventory list and shifting the remaining
79        * Cars in the inventory list to the left to fill in the gap;
80        * the method returns the Car that is being "sold" (removed from lot)
81        * and reduces the size of inventory by 1
82        * <p>
83        * PRECONDITION: indexOfCarToSell &lt; inventory.size()
84        *
85        * @param indexOfCarToSell The index of the Car to sell
86        * @return the Car that is "sold"
87        */
88       public Car sellCarShift(int indexOfCarToSell)
89       {
90           // note that the ArrayList "remove" method causes all elements to shift to the left,
91           // and conveniently returns the element that is being removed!
92           return inventory.remove(indexOfCarToSell);
93       }
94   
95       /**
96        * "Sells" the Car located at indexOfCarToSell,
97        * but instead of removing it and shifting the inventory
98        * list to the left, REPLACE the Car at indexOfCarToSell
99        * with NULL, thus creating an "empty parking spot" on the lot;
100       * this method returns the Car that is being "sold" (replaced with null),
101       * but does NOT reduce the size of the inventory
102       * <p>
103       * PRECONDITION: indexOfCarToSell &lt; inventory.size()
104       *
105       * @param indexOfCarToSell The index of the Car to sell
106       * @return the Car that is "sold"
107       */
108      public Car sellCarNoShift(int indexOfCarToSell)
109      {
110          // get the car at the requested index -- but don't remove it
111          Car soldCar = inventory.get(indexOfCarToSell);
112  
113          // update the value at the specified index to null
114          // (note this does NOT reduce the size of inventory, there is
115          // now just a null value as one of the elements!)
116          inventory.set(indexOfCarToSell, null);
117  
118          // return the sold Car
119          return soldCar;
120      }
121  
122      /**
123       * Moves Car located at index indexOfCarToMove to index destinationIndex;
124       * if destinationIndex > indexOfCarToMove, moves the Car to the right in
125       * inventory; if destinationIndex &lt; indexOfCarToMove, moves the
126       * Car to the left in the inventory. All other cars in the inventory
127       * are shifted accordingly.
128       * <p>
129       * PRECONDITIONS: indexOfCarToMove &lt; inventory.size()
130       *                destinationIndex &lt; inventory.size()
131       *
132       * @param indexOfCarToMove The index of the Car to move
133       * @param destinationIndex The index where the Car should be moved
134       */
135      public void moveCar(int indexOfCarToMove, int destinationIndex)
136      {
137          // one way to do this is to REMOVE the car from indexOfCarToMove
138          // and storing in a variable, then immediately add to destinationIndex:
139          Car carToMove = inventory.remove(indexOfCarToMove);
140          inventory.add(destinationIndex, carToMove);
141      }
142  }
